【第4节】（数学运算）

【006――加法计算1+2】

#include <stdio.h>	
#include <stdlib.h>	
#include <windows.h>
int main()
{
	int a,b,c;//int(integer)取整函数；设置a;b;c;
	a=1;//赋值a为1；“=”号编程中为赋值符号;等于号表示为“==”
	b=2;//赋值b为2；
	c=a+b;//赋值c为a+b;
	printf("%d",c);//输出结果；%d为输出实数，对应函数int;输出对象为c
	Sleep(2000);
	return 0;
}

【007――最后输出啥】

#include <stdio.h>	
#include <stdlib.h>	
#include <windows.h>
int main()
{
	int a;
	a=1;
	a=2;
	printf("%d",a);//输出2；a=1被a=2覆盖运行；始终存放最后一次的值
	Sleep(2000);
	return 0;
}

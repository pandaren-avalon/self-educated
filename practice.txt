MissionB：尝试交换1：

#include <stdio.h>	
#include<windows.h>
int main()
{
	int a,b;
	scanf_s("%d %d",&a,&b);//获取变量值a,b的输入；a=&a、b=&b
	a=b;//a=&b
	b=a;//b=a=&b

MissionC：引入临时变量t：

#include <stdio.h>

MissionE：不引入临时变量实现交换，尝试解析如何实现：

#include <stdio.h>	
#include<windows.h>
int main()
{
	int a,b;
	scanf_s("%d %d",&a,&b);
	a=b-a;//①
	b=b-a;//②
	a=b+a;//③
	printf("%d %d",a,b);
	Sleep(5000);
	return 0;
}








解析：

已经定义a,b;并获取输入值&a;&b;
	a=b-a;//①
	b=b-a;//②
	a=b+a;//③
最终输出时a的赋值①被③覆盖；类似于
int a,b,c;
a=0;
b=1;
a=2;
则最后输出(a,b)为(2,1)一个道理；


①a=b-a//a有赋值动作；	a=(&b-&a);

②b=b-a//b有赋值动作；	b=&b-a;	&b-a中有不确定变量a；查询到
	①中a=(&b-&a)
	则&b-a=&b-(&b-&a)=&b-&b+&a=&a;最后【b=&a】

③a=b+a//a有赋值动作；a={b+a}；{b+a}中有不确定变量b,a;查询到
	②中b=&a；
	①中a=(&b-&a)
	则a={b+a}={&a+(&b-&a)}={&a+&b-&a}=&b;最后【a=&b】

printf("")输出(a,b)=(&b,&a)，数据交换完成；

【032――输入三个数；输出最大的那个值】

算法1：
	1、定义4个变量a,b,c,d；d为临时变量；
	2、先比较a与b，选出较大的赋值给d；
	3、再比较c与d，选出较大的赋值给d；
	4、输出最后变量值d；

#include <stdio.h>	
#include<windows.h>
int main()
{
	int a,b,c,d;
	scanf_s("%d %d %d", &a,&b,&c);

	if (a>b)	d=a;
	else	d=b;

【第4节】（break）

#include <stdio.h>
#include<windows.h>
int main()
{
	int a, b, c;
	c = 0;
	scanf_s("%d", &a);
	for (b = 2; b <= a - 1; b++)
	{
		if (a%b == 0)
		{
			c = 1;			
			printf("%d ", b);
			break;
		}
	}
	if (c == 0)	printf("质数");
	else	printf("合数");
	Sleep(1000);
	return 0;
}

break：用来提前打断循环，如上面例子中主要算出一次约数，就打断继续循环计算
输出结果里面只要算出一个约数，直接判断该数字为合数；提高计算速度。
【总结】break：用来提前终止for、while或者do-while循环的

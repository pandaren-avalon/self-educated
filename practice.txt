#include <stdio.h>
#include <windows.h>
int main()
{
	int	a=1, b, c=0;
	while (c < 5) 
	{ b = 0; 
		while (b <= c) 
			{ printf("%d ", a); 
			b = b + 1; 
			a = a + 1;
			}
		printf("\n"); 
		c = c + 1; 
	}

	getchar();
	return 0;
}
